```kotlin
// ✅ PATTERN — Guard Clauses (Clean Code)

    // ...
    project.afterEvaluate {
        val isGradlePropertiesEnabled = bakeryExtension.configPath.isPresent
        
        project.configureConfigPath(bakeryExtension, isGradlePropertiesEnabled)
        
        // Guard 1 : configPath doit être présent
        if (!bakeryExtension.configPath.isPresent) {
            project.logger.warn("[bakery] configPath is not set...")
            registerScaffoldOnlyTasks(project, bakeryExtension, jbakeRuntime)
            return@afterEvaluate  // ← Early exit
        }
        
        // Guard 2 : vérifier fichier et répertoires
        val configFile = project.layout.projectDirectory.asFile
            .toPath()
            .resolve(bakeryExtension.configPath.get())
            .toFile()
        val configDir = configFile.parentFile
        
        if (!configFile.exists() || /* ... */) {
            project.logger.lifecycle("config file does not exist...")
            registerScaffoldOnlyTasks(project, bakeryExtension, jbakeRuntime)
        } else {
            registerFullPipelineTasks(project, bakeryExtension, jbakeRuntime, isGradlePropertiesEnabled)
        }
```

**✅ Forces** :
- ✅ **Lisibilité** : Chaque chemin logique est clair (`if not present → scaffold`, `if exists → full`)
- ✅ **Sûreté de type** : `isPresent` vérifiée AVANT `get()`
- ✅ **Timing correct** : `isGradlePropertiesEnabled` évalué DANS `afterEvaluate` (non pas avant)

**🟡 Problèmes mineurs** :

===== Problem 1 : Répétition `registerScaffoldOnlyTasks`

```kotlin

    // ... warn
    registerScaffoldOnlyTasks(...)  // ← call 1

```

**DRY violation** : 2 appels à `registerScaffoldOnlyTasks` avec contextes légèrement différents.

**Solution** (voir section 3.1) : Extraire une fonction `shouldRegisterScaffoldOnly()` qui retourne Boolean.

===== Problem 2 : Logging sans Exception Context

```kotlin

    "[bakery] configPath is not set — only scaffold tasks will be available...",
```

**Problème** : `Logger.warn(String, Throwable)` n'existe pas en Kotlin Logging. Le `Throwable` est ignoré.

**Solution** :

```kotlin
    "gradle.properties, or CLI (-Pbakery.config.path=...)"

    // something risky
```

==== Recommendation — Arrow-kt Applied

**💡 Opportunité** : Utiliser **Arrow's Raise DSL** pour la composition de guard clauses

```kotlin
// Version avec Arrow-kt (futur)
import arrow.core.raise.either
import arrow.core.raise.ensure

override fun apply(project: Project) {
    project.afterEvaluate {
        val result = either {
            val isGradlePropertiesEnabled = bakeryExtension.configPath.isPresent
            project.configureConfigPath(bakeryExtension, isGradlePropertiesEnabled)
            
            // Guard 1 : configPath présent ?
            ensure(bakeryExtension.configPath.isPresent) {
                "configPath not set. Define via DSL, gradle.properties, or -Pbakery.config.path=..."
            }
            
            // Guard 2 : fichier existe ?
            val configFile = resolveConfigFile(project, bakeryExtension)
            ensure(configFile.exists()) {
                "Config file not found: ${configFile.absolutePath}"
            }
            
            // Guard 3 : répertoires existent ?
            val configDir = configFile.parentFile
            ensure(configDir.resolve(/* srcPath */).exists()) {
                "Source directory not found"
            }
            
            // Tous les guards passés → retourner triplet
            Triple(bakeryExtension, configFile, configDir)
        }
        
        when (result) {
            is Right -> {
                val (ext, configFile, configDir) = result.value
                registerFullPipelineTasks(project, ext, configFile, configDir, ...)
            }
            is Left -> {
                project.logger.warn("[bakery] ${result.value}")
                registerScaffoldOnlyTasks(project, bakeryExtension, jbakeRuntime)
            }
        }
    }
}
```

**Avantages** :
- ✅ **Composabilité** : Guards s'empilent sans imbrication
- ✅ **Context préservé** : Erreur contient le message explicite
- ✅ **Type-safe** : `Either<String, Triple<...>>` = contrat clair
- ✅ **Pas de null** : `ensure` lève + early-exit automatiques

```kotlin
// Version avec Arrow-kt Validation (monade pour accumuler erreurs)
import arrow.core.Validated
import arrow.core.validated
import arrow.core.mapOrAccumulate

data class ResolvedConfigs(
    val firebase: FirebaseConfig,
    val googleForms: GoogleFormsConfig,
    val firebaseAuth: FirebaseAuthConfig,
    val comments: CommentsConfig,
    val analytics: AnalyticsConfig,
    val newsletter: NewsletterConfig,
    val theme: ThemeConfig,
    val layout: LayoutConfig
)

fun Project.resolveAllConfigs(
    props: Map<String, String>,
    site: SiteConfiguration,
    extension: BakeryExtension?
): Validated<List<ConfigError>, ResolvedConfigs> = mapOrAccumulate(
    ConfigResolver.resolveFirebaseConfig(props, site.firebase).validateFirebase(),
    ConfigResolver.resolveGoogleFormsConfig(props, extension?.googleForms, site.googleForms).validateGoogleForms(),
    ConfigResolver.resolveFirebaseAuthConfig(props, extension?.firebaseAuth, site.firebaseAuth).validateAuth(),
    ConfigResolver.resolveCommentsConfig(props, extension?.commentsConfig, site.comments).validateComments(),
    ConfigResolver.resolveAnalyticsConfig(props, extension?.analytics, site.analytics).validateAnalytics(),
    ConfigResolver.resolveNewsletterConfig(props, extension?.newsletter, site.newsletter).validateNewsletter(),
    ConfigResolver.resolveThemeConfig(props, extension?.theme, site.theme).validateTheme(),
    ConfigResolver.resolveLayoutConfig(props, extension?.layout, site.layout).validateLayout()
) { firebase, googleForms, auth, comments, analytics, newsletter, theme, layout ->
    ResolvedConfigs(firebase, googleForms, auth, comments, analytics, newsletter, theme, layout)
}

// Usage :
fun Project.copySiteResources(...) {
    val configs = resolveAllConfigs(props, site, extension)
    
    when (configs) {
        is Valid -> {
            val resolved = configs.value
            injectResolvedConfigIntoJbakeProperties(targetDir, site) { key, _ ->
                when (key) {
                    "firebaseApiKey" -> resolved.firebase.apiKey
                    "googleFormsFormId" -> resolved.googleForms.formId
                    // ... etc
                    else -> value
                }
            }
        }
        is Invalid -> {
            configs.value.forEach { error ->
                project.logger.warn("Config error: ${error.message}")
            }
            throw GradleException("Config validation failed")
        }
    }
}
```

**Avantages** :
- ✅ **Accumulation d'erreurs** : Si 2 configs échouent, on rapporte les 2
- ✅ **Composabilité** : Ajouter une nouvelle config = 1 ligne de mapOrAccumulate
- ✅ **Type-safe** : `ResolvedConfigs` contraint tout à la compilation

```kotlin
// Version fonctionnelle
fun loadAndEscapeAugmentedContext(
    projectDir: File,
    logger: Logger
): Either<AugmentedContextError, String> = either {
    val contextFile = projectDir.resolve("build/bakery/augmented-context.json")
    
    ensure(contextFile.exists()) {
        AugmentedContextError.FileNotFound(contextFile.absolutePath)
    }
    
    val rawJson = try {
        contextFile.readText(UTF_8)
    } catch (e: IOException) {
        raise(AugmentedContextError.ReadError(contextFile.absolutePath, e.message ?: "Unknown"))
    }
    
    escapeJsonForJavaProperties(rawJson)
}

// Usage
val contextData = loadAndEscapeAugmentedContext(projectDir, logger)
    .mapLeft { error ->
        when (error) {
            is AugmentedContextError.FileNotFound -> {
                logger.info("Augmented context not found, skipping...")
                ""
            }
            is AugmentedContextError.ReadError -> {
                logger.warn("Failed to read: ${error.details}")
                ""
            }
        }
    }
    .getOrNull() ?: ""

updateProperty(lines, "augmentedContextData", contextData)
```

**Avantages** :
- ✅ **Erreur explicite** : `AugmentedContextError` type-safe
- ✅ **Composabilité** : `.mapLeft()` pour recovery

```kotlin
// Data class pour les injections
data class LensBudgetInjections(
    val enabled: String,
    val maxArticlesPerPage: String,
    val minSimilarity: String,
    val augmentedContextData: String
)

// Fonction pure — préparation des injections
suspend fun prepareLensBudgetInjections(
    augmentedContext: AugmentedContextDsl,
    augmentedContextFile: File,
    logger: Logger
): Either<InjectionError, LensBudgetInjections> = either {
    val enabled = augmentedContext.enabled.toString()
    val maxArticles = augmentedContext.budget.maxArticlesPerPage.toString()
    val minSimilarity = augmentedContext.budget.minSimilarity.toString()
    
    val contextData = loadAndEscapeAugmentedContext(augmentedContextFile, logger)
        .mapLeft { InjectionError.AugmentedContextLoadFailed(it.toString()) }
        .getOrElse { "" }
    
    LensBudgetInjections(enabled, maxArticles, minSimilarity, contextData)
}

// Fonction pour injecter les propriétés
fun injectLensBudgetInjections(
    lines: MutableList<String>,
    injections: LensBudgetInjections
) {
    updateProperty(lines, "augmentedContextEnabled", injections.enabled)
    updateProperty(lines, "lensBudgetMaxArticlesPerPage", injections.maxArticlesPerPage)
    updateProperty(lines, "lensBudgetMinSimilarity", injections.minSimilarity)
    updateProperty(lines, "augmentedContextData", injections.augmentedContextData)
}

// Main orchestration
private fun Project.injectLensBudgetIntoJbakeProperties(
    targetDir: File,
    site: SiteConfiguration,
    augmentedContext: AugmentedContextDsl
) {
    val jbakePropsFile = targetDir.resolve(site.bake.srcPath)
        .resolve("jbake.properties")
    
    // Guard : file exists
    if (!jbakePropsFile.exists()) {
        logger.warn("jbake.properties not found at ${jbakePropsFile.absolutePath}")
        return
    }
    
    // Read + inject
    val lines = jbakePropsFile.readText(UTF_8).lines().toMutableList()
    val augmentedContextFile = projectDir.resolve("build/bakery/augmented-context.json")
    
    runBlocking {
        prepareLensBudgetInjections(augmentedContext, augmentedContextFile, logger)
            .onRight { injections ->
                injectLensBudgetInjections(lines, injections)
            }
            .onLeft { error ->
                logger.warn("Failed to prepare LENS budget injections: $error")
            }
    }
    
    jbakePropsFile.writeText(lines.joinToString("\n"), UTF_8)
    logger.lifecycle("✓ Injected LensBudget config...")
}
```

**Bénéfices** :
- ✅ **Séparation de responsabilités** : prep vs injection vs write
- ✅ **Testabilité** : `prepareLensBudgetInjections` = fonction pure testable
- ✅ **Erreurs explicites** : `Either<InjectionError, ...>` traçable

```kotlin
// Version composable (futur optionnel)
sealed class JsonPropertyEscape {
    object Newline : JsonPropertyEscape()
    object CarriageReturn : JsonPropertyEscape()
    object Tab : JsonPropertyEscape()
    object Backslash : JsonPropertyEscape()
}

fun String.escapeForJavaProperties(escapes: List<JsonPropertyEscape> = defaultEscapes): String =
    escapes.fold(this) { acc, escape ->
        when (escape) {
            JsonPropertyEscape.Newline -> acc.replace("\n", "\\n")
            JsonPropertyEscape.CarriageReturn -> acc.replace("\r", "\\r")
            JsonPropertyEscape.Tab -> acc.replace("\t", "\\t")
            JsonPropertyEscape.Backslash -> acc.replace("\\", "\\\\")
        }
    }
```

- ✅ Tester chaque escape isolément

```kotlin
sealed class SecretField {
    data class Password(val value: String) : SecretField()
    data class ApiKey(val value: String) : SecretField()
    data class Token(val value: String) : SecretField()
}

fun maskSecret(field: SecretField): String =
    when (field) {
        is SecretField.Password -> field.value.let { 
            if (it.isBlank()) "(not set)" else "***[${it.length} chars]"
        }
        is SecretField.ApiKey -> field.value.let { 
            if (it.isBlank()) "(not set)" else "***[${it.take(4)}...${it.takeLast(4)}]"
        }
        is SecretField.Token -> field.value.let { 
            if (it.isBlank()) "(not set)" else "***[JWT]"
        }
    }

internal fun buildMaskedSummary(site: SiteConfiguration): String {
    val secrets = listOf(
        "credentials.password" to maskSecret(SecretField.Password(site.pushPage.repo.credentials.password)),
        "firebase.apiKey" to maskSecret(SecretField.ApiKey(site.firebase?.project?.apiKey ?: "")),
        "github.token" to maskSecret(SecretField.Token(site.pushPage.repo.credentials.username))
    )
    
    return buildString {
        append("srcPath=$srcPath, destDir=$destDir")
```

```kotlin
// 1. Define error ADT
sealed class ConfigResolutionError {
    data class FirebaseError(val message: String) : ConfigResolutionError()
    data class GoogleFormsError(val message: String) : ConfigResolutionError()
    data class FirebaseAuthError(val message: String) : ConfigResolutionError()
    data class CommentsError(val message: String) : ConfigResolutionError()
    data class AnalyticsError(val message: String) : ConfigResolutionError()
    data class NewsletterError(val message: String) : ConfigResolutionError()
    data class ThemeError(val message: String) : ConfigResolutionError()
    data class LayoutError(val message: String) : ConfigResolutionError()
}

// 2. Wrapping resolvers return Validated
private fun validateAndResolve(
    resolverName: String,
    resolver: () -> ConfigValue
): Validated<ConfigResolutionError, ConfigValue> = 
    try {
        resolver().valid()
    } catch (e: Exception) {
        ConfigResolutionError.ParseError(
            "Failed to resolve $resolverName: ${e.message}"
        ).invalid()
    }

// 3. Compose with mapOrAccumulate
fun Project.resolveAllConfigs(
    props: Map<String, String>,
    site: SiteConfiguration,
    extension: BakeryExtension?
): Validated<NonEmptyList<ConfigResolutionError>, ResolvedConfigs> {
    return mapOrAccumulate(
        validateAndResolve("Firebase") { 
            ConfigResolver.resolveFirebaseConfig(props, site.firebase) 
        },
        validateAndResolve("GoogleForms") { 
            ConfigResolver.resolveGoogleFormsConfig(props, extension?.googleForms, site.googleForms) 
        },
        validateAndResolve("FirebaseAuth") { 
            ConfigResolver.resolveFirebaseAuthConfig(props, extension?.firebaseAuth, site.firebaseAuth) 
        },
        validateAndResolve("Comments") { 
            ConfigResolver.resolveCommentsConfig(props, extension?.commentsConfig, site.comments) 
        },
        validateAndResolve("Analytics") { 
            ConfigResolver.resolveAnalyticsConfig(props, extension?.analytics, site.analytics) 
        },
        validateAndResolve("Newsletter") { 
            ConfigResolver.resolveNewsletterConfig(props, extension?.newsletter, site.newsletter) 
        },
        validateAndResolve("Theme") { 
            ConfigResolver.resolveThemeConfig(props, extension?.theme, site.theme) 
        },
        validateAndResolve("Layout") { 
            ConfigResolver.resolveLayoutConfig(props, extension?.layout, site.layout) 
        }
    ) { firebase, googleForms, auth, comments, analytics, newsletter, theme, layout ->
        ResolvedConfigs(firebase, googleForms, auth, comments, analytics, newsletter, theme, layout)
    }
}

// 4. Usage in SiteTaskRegistrar
private fun Project.copySiteResources(...) {
    val configs = resolveAllConfigs(props, site, extension)
    
    when (configs) {
        is Valid -> {
            val resolved = configs.value
            injectResolvedConfigIntoJbakeProperties(targetDir, site) { key, _ ->
                when (key) {
                    "firebaseApiKey" -> resolved.firebase.apiKey
                    "googleFormsFormId" -> resolved.googleForms.formId
                    // ... rest
                    else -> value
                }
            }
        }
        is Invalid -> {
            val errors = configs.value.map { "  ❌ ${it.message}" }
            logger.error("Config validation failed:\n${errors.joinToString("\n")}")
            throw GradleException("Configuration resolution failed")
        }
    }
}
```

```kotlin
import arrow.core.Either
import arrow.core.Either.Left
import arrow.core.Either.Right
import arrow.core.left
import arrow.core.right
import java.io.IOException

sealed class AugmentedContextLoadError {
    data class FileNotFound(val path: String) : AugmentedContextLoadError()
    data class ReadError(val path: String, val cause: String) : AugmentedContextLoadError()
    data class ParseError(val path: String, val cause: String) : AugmentedContextLoadError()
}

fun loadAndEscapeAugmentedContext(
    projectDir: File,
    logger: Logger
): Either<AugmentedContextLoadError, String> = either {
    val contextFile = projectDir.resolve("build/bakery/augmented-context.json")
    
    // Guard 1 : file exists
    if (!contextFile.exists()) {
        return@either AugmentedContextLoadError.FileNotFound(contextFile.absolutePath).left()
    }
    
    // Guard 2 : read file
    val rawJson = try {
        contextFile.readText(UTF_8)
    } catch (e: IOException) {
        return@either AugmentedContextLoadError.ReadError(
            contextFile.absolutePath,
            e.message ?: "Unknown IO error"
        ).left()
    }
    
    // Guard 3 : validate JSON
    if (rawJson.isBlank()) {
        return@either AugmentedContextLoadError.ParseError(
            contextFile.absolutePath,
            "File is empty"
        ).left()
    }
    
    // Success
    escapeJsonForJavaProperties(rawJson).right()
}

// Usage
private fun Project.injectLensBudgetIntoJbakeProperties(...) {
    val jbakeProps = targetDir.resolve(site.bake.srcPath).resolve("jbake.properties")
    if (!jbakeProps.exists()) {
        logger.warn("jbake.properties not found...")
        return
    }
    
    val lines = jbakeProps.readText(UTF_8).lines().toMutableList()
    
    val contextData = loadAndEscapeAugmentedContext(projectDir, logger)
        .mapLeft { error ->
            when (error) {
                is AugmentedContextLoadError.FileNotFound -> {
                    logger.info("Augmented context file not found, using empty context")
                    ""
                }
                is AugmentedContextLoadError.ReadError -> {
                    logger.warn("Failed to read augmented context: ${error.cause}")
                    ""
                }
                is AugmentedContextLoadError.ParseError -> {
                    logger.warn("Augmented context file is empty: ${error.cause}")
                    ""
                }
            }
        }
        .getOrNull() ?: ""
    
    updateProperty(lines, "augmentedContextData", contextData)
    jbakeProps.writeText(lines.joinToString("\n"), UTF_8)
    logger.lifecycle("✓ Injected LensBudget config...")
}
```

**Bénéfices** :
- ✅ **Erreur explicite** : `AugmentedContextLoadError` ADT
- ✅ **Composition** : `.mapLeft()` pour recovery patterns
- ✅ **Pas d'exceptions** : Flux d'erreur fonctionnel
- ✅ **Testable** : Retour `Either` facilite mock

```kotlin
sealed class ConfigPathResolutionError {
    object GradlePropertiesDisabled : ConfigPathResolutionError()
    object GradlePropertiesFileNotFound : ConfigPathResolutionError()
    data class InvalidConfigPath(val reason: String) : ConfigPathResolutionError()
}

fun Project.resolveConfigPathFromProperties(): Either<ConfigPathResolutionError, String> = either {
    val propertiesFile = layout.projectDirectory.asFile.resolve("gradle.properties")
    
    ensure(propertiesFile.exists()) {
        ConfigPathResolutionError.GradlePropertiesFileNotFound
    }
    
    val configPath = properties[BakeryConstants.BAKERY_CONFIG_PATH_KEY]?.toString()
    
    ensure(!configPath.isNullOrBlank()) {
        ConfigPathResolutionError.InvalidConfigPath("configPath is blank")
    }
    
    ensure(configPath.isYmlUri) {
        ConfigPathResolutionError.InvalidConfigPath("configPath must be a .yml file")
    }
    
    configPath
}

fun Project.configureConfigPath(
    bakeryExtension: BakeryExtension,
    isGradlePropertiesEnabled: Boolean
) {
    if (isGradlePropertiesEnabled) return
    
    resolveConfigPathFromProperties()
        .onRight { configPath ->
            bakeryExtension.configPath.set(configPath)
            logger.lifecycle("✓ Loaded configPath from gradle.properties: $configPath")
        }
        .onLeft { error ->
            when (error) {
                is ConfigPathResolutionError.GradlePropertiesFileNotFound -> {
                    logger.info("gradle.properties not found — DSL or CLI config required")
                }
                is ConfigPathResolutionError.InvalidConfigPath -> {
                    logger.warn("Invalid configPath: ${error.reason}")
                }
                else -> {
                    logger.info("Could not resolve configPath from gradle.properties")
                }
            }
        }
}
```

**Bénéfices** :
- ✅ **Guards explicites** : `ensure()` + ADT
- ✅ **Early exit** : Pas d'imbrication
- ✅ **Logging par cas** : Chaque erreur a son message
- ✅ **Composable** : Peut retourner `Either` pour composition avec d'autres résolutions

```kotlin
import arrow.core.Reader
import arrow.core.ask

// Dependency injection via Reader
data class ConfigPromptEnvironment(
    val input: () -> String?,      // Mockable
    val output: (String) -> Unit,  // Mockable
    val logger: Logger
)

typealias ConfigPromptM<A> = Reader<ConfigPromptEnvironment, A>

fun promptForValue(
    propertyName: String,
    sensitive: Boolean = false
): ConfigPromptM<String> = ask<ConfigPromptEnvironment>().map { env ->
    val message = if (sensitive) "$propertyName (sensitive): " else "$propertyName: "
    env.output(message)
    env.input()?.takeIf { it.isNotBlank() } ?: ""
}

fun getOrPromptWithFallback(
    propertyName: String,
    cliProperty: String,
    sensitive: Boolean
): ConfigPromptM<String> = ask<ConfigPromptEnvironment>().flatMap { env ->
    val fromCli = env.logger.run { 
        // Could be injected from Gradle too
        propertyName
    }
    
    if (fromCli.isNotEmpty()) {
        Reader { env -> fromCli }
    } else {
        promptForValue(propertyName, sensitive)
    }
}

// Usage in Task
@TaskAction
fun collectSiteConfig() {
    val env = ConfigPromptEnvironment(
        input = ::readLine,
        output = ::println,
        logger = logger
    )
    
    val program = for {
        token <- getOrPromptWithFallback("GitHub Token", "githubToken", sensitive = true)
        username <- getOrPromptWithFallback("GitHub Username", "githubUsername", sensitive = false)
        repo <- getOrPromptWithFallback("Repository URL", "githubRepo", sensitive = false)
        configPath <- getOrPromptWithFallback("Config Path", "configPath", sensitive = false)
    } yield ConfigInput(token, username, repo, configPath)
    
    val input = program.run(env)
    saveConfiguration(...)
}

// Tests — Mockable!
@Test
fun `prompts for GitHub token when not in CLI`() {

        .run(env)

```
